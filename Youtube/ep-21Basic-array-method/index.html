pop(): Removes the last element from an array and returns that element. This method modifies the original array.
code
let arr = [1, 2, 3]; arr.pop(); // 3 console.log(arr); // [1, 2]

push(): Adds one or more elements to the end of an array and returns the new length of the array.

code
let arr = [1, 2]; arr.push(3); // 3 console.log(arr); // [1, 2, 3] shift(): Removes the first element from an array and returns that element. This method modifies the original array. code
let arr = [1, 2, 3]; arr.shift(); // 1 console.log(arr); // [2, 3] unshift(): Adds one or more elements to the beginning of an array and returns the new length of the array. code
let arr = [2, 3]; arr.unshift(1); // 3 console.log(arr); // [1, 2, 3]
concat(): Combines two or more arrays into a new array without modifying the original arrays. code
let arr1 = [1, 2]; let arr2 = [3, 4]; let result = arr1.concat(arr2); // [1, 2, 3, 4] console.log(arr1); // [1, 2]
indexOf(): Returns the first index at which a given element can be found in the array, or -1 if the element is not present. code
let arr = [1, 2, 3]; console.log(arr.indexOf(2)); // 1 console.log(arr.indexOf(4)); // -1
includes(): Determines whether an array includes a certain value, returning true or false. code
let arr = [1, 2, 3]; console.log(arr.includes(2)); // true console.log(arr.includes(4)); // false reverse(): Reverses the order of the elements in an array. This method modifies the original array. code
let arr = [1, 2, 3]; arr.reverse(); console.log(arr); // [3, 2, 1]
sort(): Sorts the elements of an array in place and returns the sorted array. By default, the elements are sorted as strings in ascending order. For custom sorting, you can provide a compare function. code
let arr = [3, 1, 2]; arr.sort(); // Default sorting console.log(arr); // [1, 2, 3] let numArr = [10, 1, 3]; numArr.sort((a, b) => a - b); // Custom sorting for numbers console.log(numArr); // [1, 3, 10]
slice(): Returns a shallow copy of a portion of an array into a new array. It does not modify the original array. code
let arr = [1, 2, 3, 4]; console.log(arr.slice(1, 3)); // [2, 3] console.log(arr); // [1, 2, 3, 4]
splice(): Adds or removes elements from an array, modifying the original array. It returns an array of the removed elements. code
let arr = [1, 2, 3, 4]; arr.splice(1, 2); // Removes 2 elements starting at index 1 console.log(arr); // [1, 4] let arr2 = [1, 2, 3, 4]; arr2.splice(1, 0, 'a', 'b'); // Inserts 'a' and 'b' at index 1 console.log(arr2); // [1, 'a', 'b', 2, 3, 4]
